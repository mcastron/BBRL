30  std::set<Token*, Token::pComp> tokenSet,
31  unsigned int s)
32 {
33  // For each formula size...
34  std::vector<Formula*> formulaList;
35  for (unsigned int cS = 1; cS <= s; ++cS)
36  {
37  // Initialize an iterator for 'space'
38  std::vector<std::set<Token*>::iterator> itList;
39  for (unsigned int i = 0; i < cS; ++i)
40  itList.push_back(tokenSet.begin());
41 
42  // While the first iterator is valid...
43  while (itList[0] != tokenSet.end())
44  {
45  // Push the tokens pointed by the iterators
46  std::map<std::string, Token*> tokenMap;
47  std::vector<Token*> tokenList;
48  for (unsigned int i = 0; i < cS; ++i)
49  {
50  // Generate the token to add
51  Token* token;
52  std::string symb = (*itList[i])->getSymbol();
53 
54  // Search for the token symbol in 'tokenMap'
55  std::map<std::string, Token*>::iterator it;
56 
57  // The token has already been generated during this
58  // loop
59  it = tokenMap.find(symb);
60  if (it != tokenMap.end()) { token = it->second; }
61 
62  // The token has to be generated and added to the
63  // map
64  else
65  {
66  token = Token::getToken(symb);
67  tokenMap[symb] = token;
68  }
69 
70 
71  // Add the token
72  tokenList.push_back(token);
73  }
74 
75 
76  // Try to build the corresponding formula and store it
77  // if it worked, discard the formula else
78  try
79  {
80  Formula* f = new Formula(tokenList);
81  formulaList.push_back(f);
82  }

88  // the previous one.

288 bool Formula::isValid()
289 {
290  // Ensure that tokens sharing the same symbol are sharing the same
291  // token instance
292  std::map<std::string, Token*> symbMap;
293  for (unsigned int i = 0; i < tokenList.size(); ++i)
294  {
295  std::map<std::string, Token*>::iterator it;
296  it = symbMap.find(tokenList[i]->getSymbol());
297 
298  if (it == symbMap.end())
299  symbMap[tokenList[i]->getSymbol()] = tokenList[i];
300 
301  else if (it->second != tokenList[i]) { return false; }
302  }
303 
304 
305  // Ensure that the number of outputs is equal to one
306  unsigned int nbOutputs = 0;
307  for (unsigned int i = 0; i < tokenList.size(); ++i)
308  {
309  nbOutputs += (1 - tokenList[i]->getArity());
310  if (nbOutputs < 1) { return false; }
311  }
312 
313  return (nbOutputs == 1);
314 }
