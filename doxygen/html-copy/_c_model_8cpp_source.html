224 vector<double> CModel::computeP(
225  unsigned int nX, unsigned int nU, const vector<double>& N)
226 {
227  vector<double> P(nX * nU * nX);
228  for (unsigned int x = 0; x < nX; ++x)
229  for (unsigned int u = 0; u < nU; ++u)
230  {
231  double NSum = 0.0;
232  for (unsigned int y = 0; y < nX; ++y)
233  NSum += N[nX*nU*x + nX*u + y];
234 
235  assert(NSum > 0.0);
236 
237  for (unsigned int y = 0; y < nX; ++y)
238  P[nX*nU*x + nX*u + y] = (N[nX*nU*x + nX*u + y] / NSum);
239  }
240 
241  return P;
242 }
